add_namespace = AIRDT_economy
# Event to pass partial mobilisation to every nation in the game if they don't already have it once WT hits 15%
# thanks to Ai research and tweaks mod
country_event = {
	id = AIRDT_economy.1
	title = AIRDT_economy.1.t
	desc = AIRDT_economy.1.d
	picture = Nothing

	hidden = yes
	
	fire_only_once = yes
	
	trigger = {	
		AND = {
		is_ai = yes 
		threat > 0.25
		}
	}
	
	immediate = {
		every_country = {
			limit = { 
				is_ai = yes
				OR = {
				has_idea = isolation
				has_idea = civilian_economy
				}
				NOT = { 
					has_idea = undisturbed_isolation
					has_idea = war_economy 
					has_idea = tot_economic_mobilisation 
					has_idea = partial_economic_mobilisation 
				}
			}
				
			country_event = AIRDT_economy.3		
		}
	}
	option = {
		name = AIRDT_economy.1.a
		}
}

country_event = {
	id = AIRDT_economy.9
	title = AIRDT_economy.9.t
	desc = AIRDT_economy.9.d
	hidden = yes
	mean_time_to_happen = { days = 2 }

	trigger = {
		is_ai = yes 
		OR = { AND = { has_political_power > 70 date > 1937.6.1 } AND = { has_political_power > 110 date < 1937.6.1 } }
		
		OR = { 
			NOT = { tag = CHI }
			AND = { tag = CHI OR = { has_idea = CHI_xie_juezai } }
		}
		
		NOT = { tag = JAP tag = ITA tag = ENG tag = FRA }
		
		OR = {
			AND = {
				OR = { tag = USA tag = SOV } 
				num_of_military_factories < 120
			}
			AND = {
				tag = GER
				num_of_military_factories < 90
				has_war = no
			}
			NOT = { tag = GER tag = ITA tag = JAP tag = SOV tag = ENG tag = FRA tag = USA } #minors always use free trade
		}
		
		NOT = { has_idea = free_trade }
	}
	option = { add_ideas = free_trade add_political_power = -150 }
}
