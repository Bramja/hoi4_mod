calculate_buildings_fuel_drain = {
	
	set_variable = { buildings_fuel_consumption_multiplier = 1 } # 12 fuel per civilian factory?
	if = {
		limit = {
			has_idea = civ_fuel_consumption_0
		}
		multiply_variable = { buildings_fuel_consumption_multiplier = 0.2 } # 2.4 fuel per civilian factory?
	}
	else_if = {
		limit = {
			has_idea = civ_fuel_consumption_1
		}
		multiply_variable = { buildings_fuel_consumption_multiplier = 0.5 } # 6 fuel per civilian factory?
	}
	else_if = {
		limit = {
			has_idea = civ_fuel_consumption_2
		}
	}


	
	if = {   # techs modifying the value
		limit = {
			has_tech = vehicle_industry0
		}
		#add_to_variable = { civ_fuel_usage_factor = 0.1 }
	}

    
	set_variable = { buildings_fuel_consumption_value = modifier@num_of_civilian_factories  }
	multiply_variable = { buildings_fuel_consumption_value = buildings_fuel_consumption_multiplier }
	multiply_variable = { buildings_fuel_consumption_value = -1 }
}
